N = 4;
k = 0:N-1;
function out = nmod(in, N)
out = [in(1) in(N:-1:2)];
end
function out = cs(in, N)
out = 0.5*(in + conj(nmod(in, N)));
end
function out = ca(in, N)
out = 0.5*(in - conj(nmod(in, N)));
end
